数组的添加方法

``` var arr = [1,2,3]; arr.push(4); alert(arr); //1,2,3，4 //push方法，返回值是数组的新的长度

var arr1 = [1,2,3]; arr.unshift(0); alert(arr); //0,1,2,3 IE6,IE7不支持unshift返回值 ```

数组的删除方法

``` var arr = [‘abc’,’def’,’ggg’,’hhh’,’leo’]; arr.pop(); alert(arr); //abc,def,ggg,hhh 返回删除的

var arr = [‘abc’,’def’,’ggg’,’hhh’,’leo’]; arr.shift(); alert(arr); //def,ggg,hhh,leo 返回删除的 ```

数组的splice

``` var arr = [‘abc’,’def’,’ggg’,’hhh’,’leo’]; //删除，替换，添加 arr.splice(0,1); alert(arr); //def,ggg,hhh,leo

arr.splice(0,1,’www’); alert(arr); //www,def,ggg,hhh,leo

arr.splice(1,0,’www’); alert(arr); //abc,www,def,ggg,hhh,leo ```

数组去重

var arr = [1,2,2,4,3]; for(var i=0;i<arr.length;i++){ for(var j=i+1;j<arr.length;j++){ if(arr[i] == arr[j]){ arr.splice(j,1); j--; } } } alert(arr);

sort排序

``` var arr = [‘a’,’c’,’b’]; arr.sort(); alert(arr); //a,b,c

var arr2 = [4,3,56,77,44,87]; arr2.sort(function(a,b){ return a-b; }); alert(arr2);

var arrWidth = [‘111px’,’234px’,’30px’]; arrWidth.sort(function(a,b){ return parseInt(a) - parseInt(b); }); alert(arrWidth);
